\caption{Algorithme d'Euclide}
\begin{algorithmic}[1]
\Procedure{Euclide}{$a,b$}
\Comment{PGCD de a et b}
\State $r\gets a\bmod b$
\While{$r\not=0$}\Comment{Si r = 0, on a la r√©ponse}
\State $a\gets b$
\State $b\gets r$
\State $r\gets a\bmod b$
\EndWhile
\State \Return $b$\Comment{Le PGCD est b}
\EndProcedure
\end{algorithmic}
